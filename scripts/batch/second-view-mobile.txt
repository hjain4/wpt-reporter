# This batch file contains runs testing Wikipedia the second page view for mobile.

# To run this you need to set the following environment variables:
# WMF_WPT_KEY - The secret API key for our WebPageTest instance
# STATSV_ENDPOINT - The endpoint of where we send the statsv data
# WPT_MOBILE_RUNS - The number of runs per test

# Example (make sure to change WMF_WPT_KEY)
# $ WMF_WPT_KEY=SECRET_KEY STATSV_ENDPOINT=http://localhost WPT_MOBILE_RUNS=1 node lib/index.js --batch ./scripts/batch/second-view-mobile.txt

--webPageTestKey <%WMF_WPT_KEY> --median SpeedIndex --location us-east-1:Chrome --label chrome-mobile-second --runs <%WPT_MOBILE_RUNS> --first true --emulateMobile true --endpoint <%STATSV_ENDPOINT> --sendMetrics --connectivity 3GFast --namespace webpagetest.enwiki-mobile.Facebook-second ./scripts/wptscripts/second-view-mobile.txt
